#The distribution of DCOs as a function of redshift
#I want to do it for three variations. in GSMF and ZM keep SFR fixed

Data.MSSFR.SFRprescription = 'Madau et al. (2014)'
Data.MSSFR.Zprescription   = 'MZ_GSMF'
Variations = [['Panter et al. (2004) Single', 'Langer et al. (2006)'],\
              ['Panter et al. (2004) Single', 'Ma et al. (2015)'],\
              ['Furlong et al. (2015) Double', 'Langer et al. (2006)']]

linestyles = ['-', ':', '--']

#Setting DCO types takes the longest so first loop
#over types, then variations
fig, axes = plt.subplots(1,1,figsize=(9,8))
for nrT, Type in enumerate(Types):
    print Type
    Data.COMPAS.setCOMPASDCOmask(types=Type)
    Data.COMPAS.setCOMPASData()
    Data.setBirthTimesAnd2Darrays()
    for nrV, Variation in enumerate(Variations):
        Data.MSSFR.GSMFprescription = Variation[0]
        Data.MSSFR.ZMprescription   = Variation[1]
        Data.cosmologicalIntegration()
        
        DCOperRedshift = np.sum(Data.PerSystemPerRedshift_ratesIntrinsic, axis=1)
        Redshifts      = Data.Shell_centerRedshift
        
        print 'rate %s at redshift 0 for %s %s = %s' \
              %(Type, Variation[0], Variation[1], DCOperRedshift[0])
        
        #only want a label for DCO type once when we have solid line
        if nrV == 0:
            label = Type
        else:
            label = None
        axes.plot(Redshifts, DCOperRedshift, label=label,lw=3.,\
                 c=Colours[nrT], linestyle=linestyles[nrV])
        
nameX = r'$\rm redshift $'
nameY = r'$\rm dN_{merge}\ [Gpc^{-3} yr^{-1}]  $'
axes = CV.layoutAxes(axes, nameX=nameX, nameY=nameY)
axes.legend(loc=1, prop={'size':20})
plt.tight_layout()
plt.savefig('./TotalMergerRateRedshift.png', dpi=300, transparent=False)
plt.show()
plt.close()
